/*
 1、以下程序运行后的打印结果是______(需要写上分析思路）
 #define pingfang(a) (a*a)
 #include "stdio.h"
 int main()
 {
 int c = pingfang(5+5);
 printf("c is %d\n", c);
 return 0;
 }
 */

#define pingfang(a) (a*a)
#import<Foundation/Foundation.h>
intmain(intargc,constchar* argv[])
{
   @autoreleasepool{
       intc =pingfang((5+5));
       NSLog(@"c is %d",c);
       /*
        打印出来c是35，我很奇怪,我认为是100才对的呀，原来是这样子的撒（宏采取的是直接带入）：
        
         (a)   (a*a)
         ==>
         (5+5)  (5+5*5+5)
          

           应该写成
         #define pingfang (a)   ((a)*(a))
         */
    }
    return 0;
}
